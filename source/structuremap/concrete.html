Inputs 
  src : LRoot as source 
  srcB : LRoot2 as source

Section 'Example Mappings' ----------------------------------------------------------------

root:   for src as src make tgt = create("RRoot") then by variables.
a:      for src.a as a make tgt.a = copy(a).  
as:     for src.as as as make tgt.as = truncate(as, 20).
ax:     for src.ax as ax make tgt.aj = escape(as, 'xml', 'json').
aa:     for src.aa as src_aa make tgt.aa as tgt_aa then by variables.
aa.ab:  for src_aa.ab make tgt_aa.ab = copy(ab).
b:      for src.b as b make tgt.b = cast(b).
c:      for src.c1 as c1, src.c2 as c2 make tgt.c = append(c1, ',', c2).
d:      for src.d as d make tgt.g = translate(d, "uri").
e_f:    for src.e_f as src_e_f make tgt.e as tgt_e then e_f_2.
e_f_2:  for src_e_f make tgt_e.f = copy(src_e_f), tgt_e.g = copy("g1").
e_f2:   for src.e_f2 as src_e_f2 make tgt.e as tgt_e2 {only_one, first} then by variables.
e_f2_2: for src_e_f2 make tgt_e2.f = copy(src_e_f3), tgt_e2.g = copy("g2").
h :     for src.h as h make tgt.h = translate(h, "uri").
i:      for src.i.
k:      for src.k where l = '2' make tgt.k = copy(k).
o1:     for src.m as src_m then o2.
o2:     for src_m.n as src_n then o3.
o3:     for src_n.o = o do tgt.o = copy(o).
s1:     for src.s as src_s where s.select(not t.empty) {only_one} then by variables.
s1.t:   for src_s.t as t make tgt.p = copy(t).
s2:     for src.s as src_s where s.select(not u.empty) then by variables.
s2.t:   for src_s.u as u make tgt.q = copy(u).
p:      for src.p as src_p make tgt.s_ as tgt_s then by variables.
p1:     for src_p make tgt_s.t = copy(src_p).
q:      for src.q as src_q make tgt.s_ as tgt_s then by variables.
q1:     for src_q make tgt_s.u_ = copy(src_q).
r:      for src.r as src_r make tgt.s_ as tgt_s then by variables.
r1:     for src_r make tgt_s.t_ = copy(src_r).
y1:     for src.y as y where z.inValueSet("uri1") make tgt.x = copy(y).
y2:     for src.y as y where z.inValueSet("uri2") make tgt.w = copy(y).
aza :   for src.az1 as src_az1 then azb.
azb:    for src_az1.az3 as src_az3 make tgt.az1 as tgt_az1 then az2, az3.
az2:    for src_az1.az2 as az2 make tgt_az1.az2 = copy(az2).
az3:    for src_az3 make tgt_az1.az3 = copy(src_az3).
aycode: for src.ay as ay where ay.type("code") make ayCd = copy(ay).
aycode: for src.ay as ay where ay.type("string") make aySt = copy(ay).
root1:  for srcB as srb where srcptr.refersTo(root) then root1a, root2.
root1a: for srb.f1 as f1 make tgt.f1 = copy(f1).
root2:  for src.f2 as srcff make rr = create(RRoot), tgt,ptr = reference(rr) then root2a.
root2a: for srcff make rr.f2 = copy(srcff).
